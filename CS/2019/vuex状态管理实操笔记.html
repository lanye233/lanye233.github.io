# vuex状态管理实操笔记

当你有很多参数状态需要实时改变时，你可以用vuex来管理。

# 1. 首先创建如下的目录：

└─store
    │ store.js
    │
    └─modules
            prop1Store.js
            prop2Store.js
            …

import Vue from 'vue'
import Vuex from 'vuex'
import prop1Store from './modules/prop1Store'
import prop2Store from './modules/prop2Store'

Vue.use(Vuex);

export default new Vuex.Store({
    modules: {
        prop1Store,
        prop2Store
    },
